#define RXD2 16
#define TXD2 17

#define RXD1 4
#define TXD1 2

void setup(){
  Serial.begin(115200);
  Serial1.begin(115200, SERIAL_8N1, RXD1, TXD1);
  Serial2.begin(9600, SERIAL_8N1, RXD2, TXD2);
}

void loop(){
  if(Serial1.available() > 0){
    Serial.print("Data received: ");
    // capture as an int
    int incomingByte = Serial1.read();
    Serial.println(incomingByte);
  }
}
